47 class MIfu : ::sc_core::sc_module {
48 public:
49  // Ports ...
50  sc_in<bool> clk, reset;
51 
52  // to MEMU (read port)...
53  sc_out<bool> rp_rd;

58 
59  // to EXU ...
60  sc_in<bool> next, jump, flush;
61  // (next, jump) = (1, 1) lets the (current + 1)'th instruction be the jump target.
62  // Logically, 'jump' is performed before 'next'. Hence, jump instructions may either sequentially first
63  // assert 'jump' and then 'next' or both signals in the same cycle.
64  sc_in<TWord> jump_adr;
65  sc_out<TWord> ir, pc, npc; // registered outputs
66  sc_out<bool> ir_valid, npc_valid;
67  sc_in<bool> icache_enable;
68 
69  // Constructor...
70  SC_HAS_PROCESS (MIfu);
71  MIfu (sc_module_name name)
72  : sc_module (name) {
73  SC_METHOD (OutputMethod);
74  sensitive << insn_top << adr_top << next << sig_rp_read;
75  for (int n = 0; n < CFG_IFU_IBUF_SIZE; n++) sensitive << insn_buf[n] << adr_buf[n];
76  SC_METHOD (TransitionMethod);
77  sensitive << clk.pos ();
78  }
79 
80  // Functions...
81  void Trace (sc_trace_file * tf, int levels = 1);
82 
83  // Processes...
84  void OutputMethod ();
85  void TransitionMethod ();
86 
87 protected:
88  // Registers ...
89  sc_signal<TWord> insn_buf[CFG_IFU_IBUF_SIZE];
90  sc_signal<TWord> adr_buf[CFG_IFU_IBUF_SIZE];
91  sc_signal<sc_uint<CFG_IFU_IBUF_SIZE_LD + 1> > insn_top, // 'insn_top': first buffer place with not-yet-known contents (insn)
92  adr_top; // 'adr_top': first buffer place with not-yet-known address
93 
94  sc_signal<bool> last_rp_ack;
95  sc_signal<bool> rd_ack_dirty, sig_rp_read;
96  sc_signal<TWord> sig_rp_adr;
