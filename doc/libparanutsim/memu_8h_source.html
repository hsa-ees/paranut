9  Description:
10  This is the memory unit (MEMU) of the ParaNut.
11  The MEMU interfaces with the main memory bus over a wishbone interface
12  and with the ParaNut CPUs over an arbitrary number of read and write
13  ports. The MEMU contains the (shared) cache and is optimize to handle
14  parallel memory accesses from different ports efficiently.
15  Also, the support for synchronization primitives is due to the MEMU.
16 
17  The MEMU (class 'MMemu') contains the following sub-modules:
18  - 1 tag RAM ('MTagRAM) for storing and supplying cache tag information
19  - CFG_MEMU_CACHE_BANKS cache memory banks ('MBankRam') for storing cached data
20  - 1 bus interface ('MBusIF') for the Wishbone interconnection
21  - RPORTS read ports ('MReadPort')
22  - WPORTS write ports ('MWritePort')
23  - 1 arbiter ('MArbiter') for controlling the access to the caches' tag
24  and bank data

120 template<int DWIDTH = 1, int SEL_MAX = 1>
121 struct MSelector: sc_module {
122 public:
123  // Ports ...
124  sc_in<bool> clk, reset;
125  sc_in<SSelectorIO<DWIDTH, SEL_MAX> > f_in, s_in;
126  sc_out<SSelectorIO<DWIDTH, SEL_MAX> > out;
127  sc_in<sc_uint<MAX(1, CFG_NUT_CPU_CORES_LD)> > prio;
128 
129  void TransitionMethod() {
130  if (reset == 1) {
131  SSelectorIO<DWIDTH, SEL_MAX> reg; // Default constructor is (0, 0, 0)
132  s_in_reg = reg;
133  } else
134  s_in_reg = next_s_in_reg.read ();
135  }
136 
137  void CombMethod() {
138  SSelectorIO<DWIDTH, SEL_MAX> reg, f_in_var, s_in_var, out_var;
139  bool slow_sel_valid;
140 
141  // Read inputs...
142  f_in_var = f_in.read ();
143  s_in_var = s_in.read ();
144 
145  // Set defaults...
146  reg = s_in_reg.read ();
147  slow_sel_valid = (reg.sel_valid & s_in_var.sel_valid) && (reg.sel == s_in_var.sel);
148 
149  if (prio.read () <= FAST_INDEX)
150  // Fast input has priority...
151  if (f_in_var.sel_valid)
152  out_var = f_in_var;
153  else
154  out_var = reg;
155  else
156  // Slow input has priority...
157  if (slow_sel_valid)
158  out_var = reg;
159  else
160  out_var = f_in_var;
161 
162  // Set output valid...
163  out_var.sel_valid = f_in_var.sel_valid | slow_sel_valid;
164 
165  // Write results...
166  next_s_in_reg = s_in_var;
167  out = out_var;
168  }
169 
170  SC_HAS_PROCESS(MSelector); // Necessary when not using SC_MODULE macro
171  MSelector (const sc_module_name &name, unsigned int FAST_INDEX = 0) : sc_module(name),
172  clk ("clk"), reset ("reset"), f_in ("f_in"), s_in ("s_in"), out ("out"), prio ("prio"),
173  FAST_INDEX(FAST_INDEX)
174  {
175  SC_METHOD(TransitionMethod);

177  SC_METHOD(CombMethod);

511  // Ports ...
512  sc_in<bool> clk, reset;
513 
514  // With (CPU) port...
515  // - All input ports must be held until 'port_ack' is asserted, at least for one cycle.
516  // - 'port_ack' is issued for exactly one cycle (no full handshake).
517  // - 'port_ack' is issued one cycle before the data is valid (pipelining).
518  // - The next request may be issued at earliest one cycle after 'port_ack' was asserted (which is the same cycle the data is delivered).
519  // - ('port_ack' may be asserted already in the same cycle as the request was issued.) -> minimum latency on cache hit is presently 2 clocks
520  // - If 'port_direct' is set, nothing happens with the cache (i.e. no invalidation).
521  // Hence, 'port_direct' = 0/1 should not be mixed for one adress.
522  sc_in<bool> port_rd, port_direct;
523  sc_out<bool> port_ack;
524  sc_in<TWord> port_adr;
525  sc_out<TWord> port_data;
526 
527  sc_in<bool> port_lres_scond; // for LR/SC: reserve current adress
528  sc_out<bool> port_scond_ok; // for LR/SC: set, if store-conditional would be sucessful
529 
530  // With BUSIF...
531  sc_in<TWord> busif_adr; // adress BUSIF is currently working on (to check for BUSIF hits)
532  sc_in<TWord> busif_data; // must be routed here from 'busif.data_out[]' according to 'port_adr'
533  sc_in<bool> busif_data_valid[CFG_MEMU_CACHE_BANKS];
534  sc_out<EBusIfOperation> busif_op; // This can be replaced by one bit (only 'bioReplace' is invoked)
535  sc_in<bool> busif_busy;

537  // With cache...
538  // - Adress information for tag is routed around this module from 'port_adr', for bank from 'port_adr' AND 'way_out'.
539  sc_out<bool> tag_rd, bank_rd; // 'bank_rd' and 'cache_data' must be routed according to 'port_adr'
540  sc_in<TWord> bank_data_in; // must be routed according to 'bank_sel'
541  sc_out<TWord> bank_sel;
542  sc_in<SCacheTag> tag_in;
543  sc_out<TWord> way_out;

724  // Line lock...
725  // - Implements mutex for a single cache line (tag + all data banks).
726  // - Writers are supposed to acquire a line lock.
727  // - Reader do not acquire anything => Writers must perform their actions in a safe order.
728  // - if a line lock request is held, the associated address must not change
729  // - A line lock may be released during the clock cycle that the last bank/tag access is made,
730  // given that the bank/tag lock is still held. This allows faster write cycles without write
731  // port monopolizing a line lock (keeping it permanently up).

736  // - Tag RAM must have CFG_NUT_CPU_CORES ports.
737  // - Each port is reserved for one CPU (three ports: RP #n, RP #CFG_NUT_CPU_CORES+n, WP #n).
738  // - BUSIF uses port #CFG_NUT_CPU_CORES-1.
739  // - Arbiter provides:
740  // a) Write-/Read-Lock (a write-lock excludes all readers)
741  // b) Port arbitration: Multiple readers are allowed, but only one per port (see above)
742  // c) Priority selection amoung a port: 0. BUSIF (if applicable), 1. Data read, 2. Insn Read, 3. Data Write
743  // EXAMINE: Would two ports/CPU bring speed improvement? Would a seperate port for BUSIF bring speed improvement?
744  // - The tag RAM arbitration is also used to prevent writers from replacing/changing a cache line
745  // while it is read during a cache read hit. Hence, a reader must keep its 'tagr' lock until its bank access
746  // is completed, too. This is not necessary, if a line lock is held (i.e. for cache line reading during a write miss).
